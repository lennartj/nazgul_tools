35  /**
36   * Definition of project types, including pattern matching for recognition.
37   *
38   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
39   */
40  public enum ProjectType {
41  
42      /**
43       * <p>Reactor project, of type pom. May not contain anything except module definitions.</p>
44       */
45      REACTOR(".*-reactor$", null, "pom"),
46  
47      /**
48       * <p>Parent pom project, of type pom, defining dependencies and/or build
49       * life cycles. May not contain module definitions.</p>
50       */
51      PARENT(".*-parent$", null, "pom"),
52  
53      /**
54       * <p>Pom project, defining assemblies and/or aggregation projects. May not contain module definitions.</p>
55       */
56      ASSEMBLY(".*-assembly$", null, "pom"),
57  
58      /**
59       * Aspect definition project, holding publicly available aspects.
60       */
61      ASPECT(".*-aspect$", ".*\\.aspect$", "bundle|jar"),
62  
63      /**
64       * <p>Model project defining entities. May have test-scope dependencies on test and proof-of-concept projects.</p>
65       */
66      MODEL(".*-model$", ".*\\.model$", "bundle|jar"),
67  
68      /**
69       * Application project defining JEE-deployable artifacts.
70       * Injections of implementation projects are permitted here.
71       */
72      JEE_APPLICATION(null, null, "war|ear|ejb"),
73  
74      /**
75       * Standalone application project defining runnable Java applications.
76       * Injections of implementation projects are permitted here.
77       */
78      STANDALONE_APPLICATION(".*-application$", ".*\\.application$", "bundle|jar"),
79  
80      /**
81       * <p>Example project providing runnable example code for showing the
82       * typical scenarios of the component. Should contain relevant documentation
83       * as well as cut-and-paste code. No dependency rules.</p>
84       */
85      EXAMPLE(".*-example$", ".*\\.example$", null),
86  
87      /**
88       * <p>{@code javaagent} definition project, holding implementation of a
89       * JVM agent to be launched in-process on the form
90       * <code>-javaagent:[yourpath/][agentjar].jar=[option1]=[value1],[option2]=[value2]</code></p>
91       * <p>This project type can import/inject implementation dependencies, as
92       * it is considered an application entrypoint.</p>
93       */
94      JAVA_AGENT(".*-agent$", ".*\\.agent$", "bundle|jar"),
95  
96      /**
97       * <p>API project, defining service interaction, abstract implementations and exceptions. May have compile-scope
98       * dependencies on model projects within the same component, and test-scope dependencies on test and
99       * proof-of-concept projects.</p>
100      */
101     API(".*-api$", ".*\\.api$", "bundle|jar"),
102 
103     /**
104      * <p>SPI project, defining service interaction, abstract implementations and exceptions. Must have compile-scope
105      * dependencies to API projects within the same component. May have test-scope dependencies on test and
106      * proof-of-concept projects.</p>
107      */
108     SPI(".*-spi-\\w*$", ".*\\.spi\\.\\w*$", "bundle|jar"),
109 
110     /**
111      * <p>Implementation project, implementing service interactions from an API or SPI project,
112      * including dependencies on 3rd party libraries. Must have compile-scope dependencies to API or SPI projects
113      * within the same component. May have test-scope dependencies on test and proof-of-concept projects.</p>
114      */
115     IMPLEMENTATION(".*-impl-\\w*$", ".*\\.impl\\.\\w*$", "bundle|jar"),
116 
117     /**
118      * <p>Test artifact helper project, implementing libraries facilitating testing within
119      * other projects. No dependency rules.</p>
120      */
121     TEST(".*-test$", ".*\\.test\\.\\w*$", null),
122 
123     /**
124      * <p>Integration test artifact helper project, used to perform automated
125      * tests for several projects. No dependency rules.</p>
126      */
127     INTEGRATION_TEST(".*-it$", ".*\\.it\\.\\w*$", null),
128 
129     /**
130      * <p>Codestyle helper project, providing implementations for use within the build definition cycle.
131      * Typically used within local reactors to supply changes or augmentations to build configurations
132      * such as {@code checkstyle.xml}, or custom enforcer rule implementations. No dependency rules.</p>
133      */
134     CODESTYLE(".*-codestyle$", ".*\\.codestyle$", "jar|bundle"),
135 
136     /**
137      * Project, defining a Maven plugin.
138      */
139     PLUGIN(".*-maven-plugin$", ".*\\.plugin$", "maven-plugin"),
140 
141     /**
142      * <p>Proof-of-concept helper project, holding proof of concept implementations. No dependency rules.</p>
143      */
144     PROOF_OF_CONCEPT(".*-poc$", ".*\\.poc\\.\\w*$", null);
145 
146     // Internal state
147     private Pattern artifactIdPattern;
148     private Pattern groupIdPattern;
149     private Pattern packagingPattern;
150 
151     /**
152      * ProjectType constructor.
153      *
154      * @param artifactIdPattern Maven ArtifactID regexp pattern for recognizing a ProjectType.
155      */
156     ProjectType(final String artifactIdPattern,
157                 final String groupIdPattern,
158                 final String packagingPattern) {
159 
160         if (artifactIdPattern != null) {
161             this.artifactIdPattern = Pattern.compile(artifactIdPattern);
162         }
163 
164         if (packagingPattern != null) {
165             this.packagingPattern = Pattern.compile(packagingPattern);
166         }
167 
168         if (groupIdPattern != null) {
169             this.groupIdPattern = Pattern.compile(groupIdPattern);
170         }
171     }
172 
173     /**
174      * Checks if the provided artifactID complies with the naming standard
175      * for this ProjectType.
176      *
177      * @param artifactID The artifactID which should be checked for compliance.
178      * @return {@code true} if the provided artifactID was compliant with the
179      * naming rules for this ProjectType.
180      */
181     public boolean isCompliantArtifactID(final String artifactID) {
182         return artifactID != null && (artifactIdPattern == null || artifactIdPattern.matcher(artifactID).matches());
183     }
184 
185     /**
186      * Checks if the provided groupID complies with the naming standard
187      * for this ProjectType.
188      *
189      * @param groupID The groupID which should be checked for compliance.
190      * @return {@code true} if the provided groupID was compliant with the
191      * naming rules for this ProjectType. If no groupID pattern is
192      * given, {@code true} is returned.
193      */
194     public boolean isCompliantGroupID(final String groupID) {
195         return groupID != null && (groupIdPattern == null || groupIdPattern.matcher(groupID).matches());
196     }
197 
198     /**
199      * Checks if the provided packaging complies with the standard/requirements
200      * of this ProjectType.
201      *
202      * @param packaging THe packaging which should be checked for compliance.
203      * @return {@code true} if the provided packaging was compliant with the
204      * rules for this ProjectType. If no packaging pattern was given,
205      * {@code true} is returned.
206      */
207     public boolean isCompliantPackaging(final String packaging) {
208         return packaging != null && (packagingPattern == null || packagingPattern.matcher(packaging).matches());
209     }
210 
211     /**
212      * Acquires the ProjectType instance for the provided internal Artifact,
213      * or throws an IllegalArgumentException holding an exception message
214      * if a ProjectType could not be found for the provided Artifact.
215      *
216      * @param internalArtifact The internally developed artifact
217      * @return The corresponding ProjectType.
218      */
219     public static ProjectType getProjectType(final Artifact internalArtifact) {
220 
221         Validate.notNull(internalArtifact, "Cannot handle null internalArtifact argument.");
222 
223         final List<ProjectType> matches = findCandidates(internalArtifact.getGroupId(),
224                 internalArtifact.getArtifactId(),
225                 internalArtifact.getType(),
226                 "Incorrect internalArtifact type definition for ["
227                         + internalArtifact.getGroupId()
228                         + " :: " + internalArtifact.getArtifactId()
229                         + " :: " + internalArtifact.getVersion() + "]: ");
230 
231         // All done.
232         return matches.get(0);
233     }
234 
235     /**
236      * Acquires the ProjectType instance for the provided MavenProject,
237      * or throws an IllegalArgumentException holding an exception message
238      * if a ProjectType could not be found for the provided MavenProject.
239      *
240      * @param project The MavenProject to classify.
241      * @return The corresponding ProjectType.
242      * @throws IllegalArgumentException if the given project could not be mapped to a [single] ProjectType.
243      *                                  The exception message holds
244      */
245     public static ProjectType getProjectType(final MavenProject project) throws IllegalArgumentException {
246 
247         Validate.notNull(project, "Cannot handle null project argument.");
248 
249         final List<ProjectType> matches = findCandidates(project.getGroupId(),
250                 project.getArtifactId(),
251                 project.getPackaging(),
252                 "Incorrect project type definition for ["
253                         + project.getGroupId()
254                         + " :: " + project.getArtifactId()
255                         + " :: " + project.getVersion() + "]: ");
256 
257         // Validate the internal requirements for the two different pom projects.
258         final ProjectType toReturn = matches.get(0);
259         switch (toReturn) {
260             case PARENT:
261             case ASSEMBLY:
262                 // This project should not contain modules.
263                 if (project.getModules() != null && !project.getModules().isEmpty()) {
264                     throw new IllegalArgumentException(ProjectType.PARENT
265                             + " projects may not contain module definitions. "
266                             + "(Modules are reserved for reactor projects).");
267                 }
268                 break;
269 
270             case REACTOR:
271                 // This project not contain dependency definitions.
272                 final List<?> dependencies = project.getDependencies();
273                 if (dependencies != null && !dependencies.isEmpty()) {
274                     throw new IllegalArgumentException(ProjectType.REACTOR
275                             + " projects may not contain dependency definitions."
276                             + " (Dependencies should be defined within parent projects).");
277                 }
278 
279                 final DependencyManagement dependencyManagement = project.getDependencyManagement();
280                 if (dependencyManagement != null) {
281 
282                     // Dig out all dependency management-defined dependencies.
283                     final List<Dependency> templateDependencies = dependencyManagement.getDependencies();
284                     if (templateDependencies != null && !templateDependencies.isEmpty()) {
285                         throw new IllegalArgumentException(ProjectType.REACTOR
286                                 + " projects may not contain dependency [management] definitions."
287                                 + " (Dependencies should be defined within parent projects).");
288                     }
289                 }
290                 break;
291 
292             // No action should be taken for other project types.
293             default:
294                 break;
295         }
296 
297         // All done.
298         return toReturn;
299     }
300 
301     //
302     // Private helpers
303     //
304 
305     private static List<ProjectType> findCandidates(final String groupId,
306                                                     final String artifactId,
307                                                     final String packaging,
308                                                     final String prefix) {
309 
310         final List<ProjectType> matches = new ArrayList<ProjectType>();
311 
312         for (final ProjectType current : ProjectType.values()) {
313             if (current.isCompliantArtifactID(artifactId)
314                     && current.isCompliantGroupID(groupId)
315                     && current.isCompliantPackaging(packaging)) {
316                 matches.add(current);
317             }
318         }
319 
320         // Check sanity
321         if (matches.isEmpty()) {
322             throw new IllegalArgumentException(prefix + " Not matching any defined project types.");
323         }
324         if (matches.size() > 1) {
325             throw new IllegalArgumentException(prefix + " Matching several project types (" + matches + ").");
326         }
327         return matches;
328     }
