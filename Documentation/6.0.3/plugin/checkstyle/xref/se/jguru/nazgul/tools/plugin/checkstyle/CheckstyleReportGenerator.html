110         String title;
111 
112         if (getSeverityLevel() == null) {
113             title = bundle.getString("report.checkstyle.title");

121     public void generateReport(CheckstyleResults results) {
122         doHeading();
123 
124         if (getSeverityLevel() == null) {
125             if (enableSeveritySummary) {
126                 doSeveritySummary(results);
127             }
128 
129             if (enableFilesSummary) {
130                 doFilesSummary(results);
131             }
132 
133             if (enableRulesSummary) {
134                 doRulesSummary(results);
135             }
136         }
137 
138         doDetails(results);
139         sink.body_();
140         sink.flush();
141         sink.close();
142     }

144     private void doHeading() {
145         sink.head();
146         sink.title();
147         sink.text(getTitle());
148         sink.title_();
149         sink.head_();
150 
151         sink.body();
152 
153         sink.section1();
154         sink.sectionTitle1();
155         sink.text(getTitle());
156         sink.sectionTitle1_();
157 
158         sink.paragraph();
159         sink.text(bundle.getString("report.checkstyle.checkstylelink") + " ");
160         sink.link("http://checkstyle.sourceforge.net/");
161         sink.text("Checkstyle");
162         sink.link_();
163         String version = getCheckstyleVersion();
164         if (version != null) {
165             sink.text(" ");
166             sink.text(version);
167         }

169         sink.text(String.format(bundle.getString("report.checkstyle.ruleset"), ruleset));
170         sink.text(".");

172         if (enableRSS) {
173             sink.nonBreakingSpace();
174             sink.link("checkstyle.rss");
175             sink.figure();
176             sink.figureCaption();
177             sink.text("rss feed");
178             sink.figureCaption_();
179             sink.figureGraphics("images/rss.png");
180             sink.figure_();
181             sink.link_();
182         }
183 
184         sink.paragraph_();
185         sink.section1_();
186     }

188     /**
189      * Get the value of the specified attribute from the Checkstyle configuration.
190      * If parentConfigurations is non-null and non-empty, the parent
191      * configurations are searched if the attribute cannot be found in the
192      * current configuration. If the attribute is still not found, the
193      * specified default value will be returned.
194      *
195      * @param config              The current Checkstyle configuration
196      * @param parentConfiguration The configuration of the parent of the current configuration
197      * @param attributeName       The name of the attribute
198      * @param defaultValue        The default value to use if the attribute cannot be found in any configuration
199      * @return The value of the specified attribute
200      */
201     private String getConfigAttribute(Configuration config, ChainedItem<Configuration> parentConfiguration,
202                                       String attributeName, String defaultValue) {
203         String ret;
204         try {
205             ret = config.getAttribute(attributeName);
206         } catch (CheckstyleException e) {
207             // Try to find the attribute in a parent, if there are any
208             if (parentConfiguration != null) {
209                 ret =
210                         getConfigAttribute(parentConfiguration.value, parentConfiguration.parent, attributeName,
211                                 defaultValue);
212             } else {
213                 ret = defaultValue;
214             }
215         }
216         return ret;
217     }

250         sink.text(bundle.getString("report.checkstyle.column.severity"));
251         sink.tableHeaderCell_();

276     /**
277      * Create a summary for one Checkstyle rule.
278      *
279      * @param ref              The configuration reference for the row
280      * @param results          The results to summarize
281      * @param previousCategory The previous row's category
282      */
283     private void doRuleRow(ConfReference ref, CheckstyleResults results, String previousCategory) {
284         Configuration checkerConfig = ref.configuration;
285         ChainedItem<Configuration> parentConfiguration = ref.parentConfiguration;
286         String ruleName = checkerConfig.getName();
287 
288         sink.tableRow();
289 
290         // column 1: rule category
291         sink.tableCell();
292         String category = ref.category;
293         if (!category.equals(previousCategory)) {
294             sink.text(category);
295         }
296         sink.tableCell_();
297 
298         // column 2: Rule name + configured attributes
299         sink.tableCell();
300         if (!"extension".equals(category)) {
301             sink.link("http://checkstyle.sourceforge.net/config_" + category + ".html#" + ruleName);
302             sink.text(ruleName);
303             sink.link_();
304         } else {
305             sink.text(ruleName);
306         }
307 
308         List<String> attribnames = new ArrayList<>(Arrays.asList(checkerConfig.getAttributeNames()));
309         attribnames.remove("severity"); // special value (deserves unique column)
310         if (!attribnames.isEmpty()) {
311             sink.list();
312             for (String name : attribnames) {
313                 sink.listItem();
314 
315                 sink.text(name);
316 
317                 String value = getConfigAttribute(checkerConfig, null, name, "");
318                 // special case, Header.header and RegexpHeader.header
319                 if ("header".equals(name) && ("Header".equals(ruleName) || "RegexpHeader".equals(ruleName))) {
320                     String[] lines = StringUtils.split(value, "\\n");
321                     int linenum = 1;
322                     for (String line : lines) {
323                         sink.lineBreak();
324                         sink.rawText("<span style=\"color: gray\">");
325                         sink.text(linenum + ":");
326                         sink.rawText("</span>");
327                         sink.nonBreakingSpace();
328                         sink.monospaced();
329                         sink.text(line);
330                         sink.monospaced_();
331                         linenum++;
332                     }
333                 } else if ("headerFile".equals(name) && "RegexpHeader".equals(ruleName)) {
334                     sink.text(": ");
335                     sink.monospaced();
336                     sink.text("\"");
337                     if (basedir != null) {
338                         // Make the headerFile value relative to ${basedir}
339                         String path = siteTool.getRelativePath(value, basedir.getAbsolutePath());
340                         sink.text(path.replace('\\', '/'));
341                     } else {
342                         sink.text(value);
343                     }
344                     sink.text("\"");
345                     sink.monospaced_();
346                 } else {
347                     sink.text(": ");
348                     sink.monospaced();
349                     sink.text("\"");
350                     sink.text(value);
351                     sink.text("\"");
352                     sink.monospaced_();
353                 }
354                 sink.listItem_();
355             }
356             sink.list_();
357         }
358 
359         sink.tableCell_();
360 
361         // column 3: rule violation count
362         sink.tableCell();
363         sink.text(String.valueOf(ref.violations));
364         sink.tableCell_();
365 
366         // column 4: severity
367         sink.tableCell();
368         // Grab the severity from the rule configuration, this time use error as default value
369         // Also pass along all parent configurations, so that we can try to find the severity there
370         String severity = getConfigAttribute(checkerConfig, parentConfiguration, "severity", "error");
371         iconTool.iconSeverity(severity, IconTool.TEXT_SIMPLE);
372         sink.tableCell_();
373 
374         sink.tableRow_();
375     }

377     /**
378      * Check if a violation matches a rule.
379      *
380      * @param event            the violation to check
381      * @param ruleName         The name of the rule
382      * @param expectedMessage  A message that, if it's not null, will be matched to the message from the violation
383      * @param expectedSeverity A severity that, if it's not null, will be matched to the severity from the violation
384      * @return The number of rule violations
385      */
386     public boolean matchRule(AuditEvent event, String ruleName, String expectedMessage, String expectedSeverity) {
387         if (!ruleName.equals(RuleUtil.getName(event))) {
388             return false;
389         }
390 
391         // check message too, for those that have a specific one.
392         // like GenericIllegalRegexp and Regexp
393         if (expectedMessage != null) {
394             // event.getMessage() uses java.text.MessageFormat in its implementation.
395             // Read MessageFormat Javadoc about single quote:
396             // http://java.sun.com/j2se/1.4.2/docs/api/java/text/MessageFormat.html
397             String msgWithoutSingleQuote = StringUtils.replace(expectedMessage, "'", "");
398 
399             return expectedMessage.equals(event.getMessage()) || msgWithoutSingleQuote.equals(event.getMessage());
400         }
401         // Check the severity. This helps to distinguish between
402         // different configurations for the same rule, where each
403         // configuration has a different severity, like JavadocMetod.
404         // See also http://jira.codehaus.org/browse/MCHECKSTYLE-41
405         if (expectedSeverity != null) {
406             return expectedSeverity.equals(event.getSeverityLevel().getName());
407         }
408 
409         return true;
410     }

492             sink.tableRow();

495             sink.link("#" + filename.replace('/', '.'));
496             sink.text(filename);
497             sink.link_();
498             sink.tableCell_();

501             sink.text(String.valueOf(results.getSeverityCount(violations, SeverityLevel.INFO)));

519     private void doDetails(CheckstyleResults results) {
520 
521         sink.section1();
522         sink.sectionTitle1();
523         sink.text(bundle.getString("report.checkstyle.details"));
524         sink.sectionTitle1_();
525 
526         // Sort the files before writing their details to the report
527         List<String> fileList = new ArrayList<>(results.getFiles().keySet());
528         Collections.sort(fileList);
529 
530         for (String file : fileList) {
531             List<AuditEvent> violations = results.getFileViolations(file);
532 
533             if (violations.isEmpty()) {
534                 // skip files without violations
535                 continue;
536             }
537 
538             sink.section2();
539             SinkEventAttributes attrs = new SinkEventAttributeSet();
540             attrs.addAttribute(SinkEventAttributes.ID, file.replace('/', '.'));
541             sink.sectionTitle(Sink.SECTION_LEVEL_2, attrs);
542             sink.text(file);
543             sink.sectionTitle_(Sink.SECTION_LEVEL_2);
544 
545             sink.table();
546             sink.tableRow();
547             sink.tableHeaderCell();
548             sink.text(bundle.getString("report.checkstyle.column.severity"));
549             sink.tableHeaderCell_();
550             sink.tableHeaderCell();
551             sink.text(bundle.getString("report.checkstyle.rule.category"));
552             sink.tableHeaderCell_();
553             sink.tableHeaderCell();
554             sink.text(bundle.getString("report.checkstyle.rule"));
555             sink.tableHeaderCell_();
556             sink.tableHeaderCell();
557             sink.text(bundle.getString("report.checkstyle.column.message"));
558             sink.tableHeaderCell_();
559             sink.tableHeaderCell();
560             sink.text(bundle.getString("report.checkstyle.column.line"));
561             sink.tableHeaderCell_();
562             sink.tableRow_();
563 
564             doFileEvents(violations, file);
565 
566             sink.table_();
567             sink.section2_();
568         }
569 
570         sink.section1_();
571     }

573     private void doFileEvents(List<AuditEvent> eventList, String filename) {
574         for (AuditEvent event : eventList) {
575             SeverityLevel level = event.getSeverityLevel();
576 
577             if ((getSeverityLevel() != null) && !(getSeverityLevel() != level)) {
578                 continue;
579             }
580 
581             sink.tableRow();
582 
583             sink.tableCell();
584             iconTool.iconSeverity(level.getName(), IconTool.TEXT_SIMPLE);
585             sink.tableCell_();
586 
587             sink.tableCell();
588             String category = RuleUtil.getCategory(event);
589             if (category != null) {
590                 sink.text(category);
591             }
592             sink.tableCell_();
593 
594             sink.tableCell();
595             String ruleName = RuleUtil.getName(event);
596             if (ruleName != null) {
597                 sink.text(ruleName);
598             }
599             sink.tableCell_();
600 
601             sink.tableCell();
602             sink.text(event.getMessage());
603             sink.tableCell_();
604 
605             sink.tableCell();
606 
607             int line = event.getLine();
608             if (getXrefLocation() != null && line != 0) {
609                 sink.link(getXrefLocation() + "/" + filename.replaceAll("\\.java$", ".html") + "#L"
610                         + line);
611                 sink.text(String.valueOf(line));
612                 sink.link_();
613             } else if (line != 0) {
614                 sink.text(String.valueOf(line));
615             }
616             sink.tableCell_();
617 
618             sink.tableRow_();
619         }
620     }

697     public List<ConfReference> sortConfiguration(CheckstyleResults results) {
698         List<ConfReference> result = new ArrayList<>();
699 
700         sortConfiguration(result, checkstyleConfig, null, results);
701 
702         Collections.sort(result);

707     private void sortConfiguration(List<ConfReference> result, Configuration config,
708                                    ChainedItem<Configuration> parent, CheckstyleResults results) {
709         for (Configuration childConfig : config.getChildren()) {
710             String ruleName = childConfig.getName();
711 
712             if (treeWalkerNames.contains(ruleName)) {
713                 // special sub-case: TreeWalker is the parent of multiple rules, not an effective rule
714                 sortConfiguration(result, childConfig, new ChainedItem<>(config, parent), results);
715             } else {
716                 String fixedmessage = getConfigAttribute(childConfig, null, "message", null);
717                 // Grab the severity from the rule configuration, use null as default value
718                 String configSeverity = getConfigAttribute(childConfig, null, "severity", null);
719 
720                 // count rule violations
721                 long violations = 0;
722                 AuditEvent lastMatchedEvent = null;
723                 for (List<AuditEvent> errors : results.getFiles().values()) {
724                     for (AuditEvent event : errors) {
725                         if (matchRule(event, ruleName, fixedmessage, configSeverity)) {
726                             lastMatchedEvent = event;
727                             violations++;
728                         }
729                     }
730                 }
731 
732                 if (violations > 0) // forget rules without violations
733                 {
734                     String category = RuleUtil.getCategory(lastMatchedEvent);
735 
736                     result.add(new ConfReference(category, childConfig, parent, violations, result.size()));
737                 }
738             }
739         }
740     }

742     private static class ConfReference
743             implements Comparable<ConfReference> {
744         private final String category;
745         private final Configuration configuration;
746         private final ChainedItem<Configuration> parentConfiguration;
747         private final long violations;
748         private final int count;
749 
750         public ConfReference(String category, Configuration configuration,
751                              ChainedItem<Configuration> parentConfiguration, long violations, int count) {
752             this.category = category;
753             this.configuration = configuration;
754             this.parentConfiguration = parentConfiguration;
755             this.violations = violations;
756             this.count = count;
757         }
758 
759         public int compareTo(ConfReference o) {
760             int compare = category.compareTo(o.category);
761             if (compare == 0) {
762                 compare = configuration.getName().compareTo(o.configuration.getName());
763             }
764             return (compare == 0) ? (o.count - count) : compare;
765         }
766     }
