23  package se.jguru.nazgul.tools.codestyle.enforcer.rules;
24  
25  import org.apache.maven.enforcer.rule.api.EnforcerRuleHelper;
26  import org.apache.maven.project.MavenProject;
27  
28  /**
29   * Enforcer rule to validate ProjectType compliance, to harmonize the pom structure in terms
30   * of groupId, artifactId, packaging and (rudimentary) content checks.
31   *
32   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
33   */
34  public class ValidateProjectType extends AbstractNonCacheableEnforcerRule {
35  
36      /**
37       * Delegate method, implemented by concrete subclasses.
38       *
39       * @param project The active MavenProject.
40       * @param helper  The EnforcerRuleHelper instance, from which the MavenProject has been retrieved.
41       * @throws se.jguru.nazgul.tools.codestyle.enforcer.rules.RuleFailureException If the enforcer rule
42       *                                                                             was not satisfied.
43       */
44      @Override
45      @SuppressWarnings("PMD.PreserveStackTrace")
46      protected void performValidation(final MavenProject project, final EnforcerRuleHelper helper)
47              throws RuleFailureException {
48  
49          try {
50              ProjectType.getProjectType(project);
51          } catch (IllegalArgumentException e) {
52              throw new RuleFailureException(e.getMessage());
53          }
54      }
55  
56      /**
57       * @return A human-readable short description for this AbstractEnforcerRule.
58       * (Example: "No -impl dependencies permitted in this project")
59       */
60      @Override
61      protected String getShortRuleDescription() {
62          return "POM groupId, artifactId and packaging must comply with defined standard";
63      }
64  }
