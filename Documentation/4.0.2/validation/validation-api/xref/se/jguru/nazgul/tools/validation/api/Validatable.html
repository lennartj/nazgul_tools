23  package se.jguru.nazgul.tools.validation.api;
24  
25  import se.jguru.nazgul.tools.validation.api.exception.InternalStateValidationException;
26  
27  /**
28   * Specification for providing an object with Validation mechanics.
29   * Each Validatable object can perform <strong>rudimentary</strong>
30   * validation to assert its state before being serialized and after
31   * being deserialized.
32   * <p/>
33   * Making an object implement Validatable does not imply that all
34   * uses of the object is guaranteed. Validatable objects should
35   * primarily make use of their own data to ascertain its valid state.
36   * <p/>
37   * It is the responsibilities of services using the Validatable object
38   * (as opposed to the validation mechanics provided within this Validatable)
39   * to provide extra/semantic validation for object <strong>graphs</strong>
40   * in which this Validatable instance is part.
41   *
42   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
43   */
44  public interface Validatable {
45  
46      /**
47       * Performs validation of the internal state of this Validatable.
48       *
49       * @throws InternalStateValidationException
50       *          if the state of this Validatable was
51       *          in an incorrect state (i.e. invalid).
52       */
53      void validateInternalState() throws InternalStateValidationException;
54  }
