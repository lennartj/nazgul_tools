23  package se.jguru.nazgul.tools.validation.api.exception;
24  
25  import se.jguru.nazgul.tools.validation.api.expression.ExpressionBuilder;
26  
27  /**
28   * Exception indicating problems occurred when validating a Validatable instance.
29   *
30   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
31   */
32  public class InternalStateValidationException extends AbstractErrorMessageContainer {
33  
34      // Internal state
35      private static final long serialVersionUID = 2494787185488486468L;
36  
37      /**
38       * Factory method to create and return an ExpressionBuilder holding an empty
39       * InternalStateValidationException as MessageContainer.
40       *
41       * @return A newly created InternalStateValidationException instance with an empty message buffer.
42       */
43      public static ExpressionBuilder create() {
44          return new ExpressionBuilder(new InternalStateValidationException());
45      }
46  
47      /**
48       * Factory method to create and return an ExpressionBuilder holding an
49       * InternalStateValidationException (given the provided cause) as MessageContainer.
50       *
51       * @param cause the cause (which is saved for later retrieval by the
52       *              {@link Throwable#getCause()} method).  (A <tt>null</tt> value is
53       *              permitted, and indicates that the cause is nonexistent or
54       *              unknown.)
55       * @return A newly created InternalStateValidationException instance with an empty message buffer.
56       */
57      public static ExpressionBuilder create(final Throwable cause) {
58          return new ExpressionBuilder(new InternalStateValidationException(cause));
59      }
60  
61      /**
62       * Constructs an IllegalStateException with no detail message.
63       * A detail message is a String that describes this particular exception.
64       */
65      public InternalStateValidationException() {
66      }
67  
68      /**
69       * Constructs a new exception with the specified cause and a detail
70       * message of <tt>(cause==null ? null : cause.toString())</tt> (which
71       * typically contains the class and detail message of <tt>cause</tt>).
72       * This constructor is useful for exception that are little more than
73       * wrappers for other throwables (for example, {@link
74       * java.security.PrivilegedActionException}).
75       *
76       * @param cause the cause (which is saved for later retrieval by the
77       *              {@link Throwable#getCause()} method).  (A <tt>null</tt> value is
78       *              permitted, and indicates that the cause is nonexistent or
79       *              unknown.)
80       * @since 1.5
81       */
82      private InternalStateValidationException(final Throwable cause) {
83          super(cause);
84      }
85  }
