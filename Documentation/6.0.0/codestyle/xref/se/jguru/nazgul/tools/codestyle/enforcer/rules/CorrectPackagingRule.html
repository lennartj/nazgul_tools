22  package se.jguru.nazgul.tools.codestyle.enforcer.rules;
23  
24  import org.apache.commons.lang.Validate;
25  import org.apache.maven.enforcer.rule.api.EnforcerRuleHelper;
26  import org.apache.maven.project.MavenProject;
27  import se.jguru.nazgul.tools.codestyle.enforcer.rules.source.JavaPackageExtractor;
28  import se.jguru.nazgul.tools.codestyle.enforcer.rules.source.PackageExtractor;
29  
30  import java.io.File;
31  import java.io.FileFilter;
32  import java.util.ArrayList;
33  import java.util.Arrays;
34  import java.util.List;
35  import java.util.Map;
36  import java.util.SortedMap;
37  import java.util.SortedSet;
38  import java.util.TreeMap;
39  import java.util.TreeSet;
40  
41  /**
42   * Enforcer rule to enforce correct packaging for all source files within a project,
43   * implying that all source files should be located within or under a package identical
44   * to the groupId of the project itself.
45   *
46   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
47   */
48  public class CorrectPackagingRule extends AbstractNonCacheableEnforcerRule {
49  
50      // Constants
51      private static final FileFilter DIRECTORY_FILTER = new FileFilter() {
52          @Override
53          public boolean accept(final File candidate) {
54              return candidate.isDirectory();
55          }
56      };
57  
58      /**
59       * The default List of PackageExtractors used to identify packages within found source files.
60       */
61      public static final List<PackageExtractor> DEFAULT_PACKAGE_EXTRACTORS = Arrays.asList(
62              (PackageExtractor) new JavaPackageExtractor());
63  
64      // Internal state
65      private List<PackageExtractor> packageExtractors;
66  
67      /**
68       * Default constructor, assigning the {@code DEFAULT_PACKAGE_EXTRACTORS} for
69       * use as all known PackageExtractor instances.
70       */
71      public CorrectPackagingRule() {
72  
73          // Use the default package extractor List
74          packageExtractors = DEFAULT_PACKAGE_EXTRACTORS;
75      }
76  
77      /**
78       * Delegate method, implemented by concrete subclasses.
79       *
80       * @param project The active MavenProject.
81       * @param helper  The EnforcerRuleHelper instance, from which the MavenProject has been retrieved.
82       * @throws RuleFailureException If the enforcer rule was not satisfied.
83       */
84      @Override
85      @SuppressWarnings("unchecked")
86      protected void performValidation(final MavenProject project, final EnforcerRuleHelper helper)
87              throws RuleFailureException {
88  
89          // Find all java source files, and map their packages to their names.
90          final List<String> compileSourceRoots = (List<String>) project.getCompileSourceRoots();
91          if (compileSourceRoots.size() == 0) {
92              return;
93          }
94  
95          final SortedMap<String, SortedSet<String>> packageName2SourceFileNameMap
96                  = new TreeMap<String, SortedSet<String>>();
97  
98          for (String current : compileSourceRoots) {
99              addPackages(new File(current), packageName2SourceFileNameMap);
100         }
101 
102         // Retrieve the groupId of this project
103         final String groupId = project.getGroupId();
104         if (groupId == null || groupId.equals("")) {
105 
106             // Don't accept empty groupIds
107             throw new RuleFailureException("Maven groupId cannot be null or empty.", project.getArtifact());
108 
109         } else {
110 
111             // Correct packaging everywhere?
112             final SortedSet<String> incorrectPackages = new TreeSet<String>();
113             for (Map.Entry<String, SortedSet<String>> currentPackage : packageName2SourceFileNameMap.entrySet()) {
114 
115                 final String candidate = currentPackage.getKey();
116                 if (!candidate.startsWith(groupId)) {
117                     incorrectPackages.add(candidate);
118                 }
119             }
120 
121             if (incorrectPackages.size() > 0) {
122 
123                 final SortedMap<String, SortedSet<String>> result = new TreeMap<String, SortedSet<String>>();
124                 for (String current : incorrectPackages) {
125                     result.put(current, packageName2SourceFileNameMap.get(current));
126                 }
127 
128                 throw new RuleFailureException("Incorrect packaging detected; required [" + groupId
129                         + "] but found package to file names: " + result, project.getArtifact());
130             }
131         }
132     }
133 
134     /**
135      * {@inheritDoc}
136      */
137     @Override
138     protected String getShortRuleDescription() {
139         return "Topmost source package must be identical to project groupId.";
140     }
141 
142     /**
143      * Splices the supplied packageExtractorImplementations argument, which is assumed to be a comma-separated
144      * string holding fully qualified class names of the PackageExtractor implementations which should be used
145      * by this CorrectPackagingRule.
146      *
147      * @param packageExtractorImplementations a comma-separated string holding fully qualified class names of the
148      *                                        PackageExtractor implementations. Each such class must have a default
149      *                                        (i.e. no-argument) constructor.
150      * @throws IllegalArgumentException if the supplied packageExtractorImplementations argument could not yield an
151      *                                  instantiated PackageExtractor instance.
152      */
153     public final void setPackageExtractors(final String packageExtractorImplementations)
154             throws IllegalArgumentException {
155 
156         // Check sanity
157         Validate.notEmpty(packageExtractorImplementations,
158                 "Cannot handle empty packageExtractorImplementations argument.");
159 
160         // Instantiate the PackageExtractor instances.
161         List<PackageExtractor> extractors = new ArrayList<PackageExtractor>();
162         for (String current : splice(packageExtractorImplementations)) {
163             try {
164 
165                 // Load the current PackageExtractor implementation class
166                 final Class<?> aClass = getClass().getClassLoader().loadClass(current);
167 
168                 // The PackageExtractor implementation must have a default constructor.
169                 // Fire, and handle any exceptions.
170                 extractors.add((PackageExtractor) aClass.newInstance());
171             } catch (Exception e) {
172                 throw new IllegalArgumentException("Could not instantiate PackageExtractor from class ["
173                         + current + "]. Validate that implementation has a default constructor, and implements the"
174                         + PackageExtractor.class.getSimpleName() + " interface.");
175             }
176         }
177 
178         // Assign if non-null.
179         if (extractors.size() > 0) {
180             this.packageExtractors = extractors;
181         }
182     }
183 
184     //
185     // Private helpers
186     //
187 
188     /**
189      * Adds all source file found by recursive search under sourceRoot to the
190      * toPopulate List, using a width-first approach.
191      *
192      * @param fileOrDirectory      The file or directory to search for packages and [if a directory] recursively
193      *                             search for further source files.
194      * @param package2FileNamesMap A Map relating package names extracted by the PackageExtractors.
195      */
196     private void addPackages(final File fileOrDirectory,
197                              final SortedMap<String, SortedSet<String>> package2FileNamesMap) {
198 
199         for (PackageExtractor current : packageExtractors) {
200 
201             final FileFilter sourceFileDefinitionFilter = current.getSourceFileFilter();
202 
203             if (fileOrDirectory.isFile() && sourceFileDefinitionFilter.accept(fileOrDirectory)) {
204 
205                 // Single source file to add.
206                 final String thePackage = current.getPackage(fileOrDirectory);
207 
208                 SortedSet<String> sourceFileNames = package2FileNamesMap.get(thePackage);
209                 if (sourceFileNames == null) {
210                     sourceFileNames = new TreeSet<String>();
211                     package2FileNamesMap.put(thePackage, sourceFileNames);
212                 }
213 
214                 // Done.
215                 sourceFileNames.add(fileOrDirectory.getName());
216 
217             } else if (fileOrDirectory.isDirectory()) {
218 
219                 // Add the immediate source files
220                 for (File currentChild : fileOrDirectory.listFiles(sourceFileDefinitionFilter)) {
221                     addPackages(currentChild, package2FileNamesMap);
222                 }
223 
224                 // Recurse into subdirectories
225                 for (File currentSubdirectory : fileOrDirectory.listFiles(DIRECTORY_FILTER)) {
226                     addPackages(currentSubdirectory, package2FileNamesMap);
227                 }
228             }
229         }
230     }
