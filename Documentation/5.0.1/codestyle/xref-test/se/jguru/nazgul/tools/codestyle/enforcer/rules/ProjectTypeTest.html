72      @Test
73      public void validateParsingProjectTypes() {
74  
75          // Assemble
76          final MavenProject parent = MavenTestUtils.getStub("pom", "se.jguru.foo.bar", "bar-parent");
77          final MavenProject reactor = MavenTestUtils.getStub("pom", "se.jguru.foo.bar", "bar-reactor");
78          final MavenProject assembly = MavenTestUtils.getStub("pom", "se.jguru.foo.bar", "bar-assembly");
79          final MavenProject model = MavenTestUtils.getStub("bundle", "se.jguru.foo.bar.model", "bar-model");
80          final MavenProject api = MavenTestUtils.getStub("bundle", "se.jguru.foo.bar.api", "bar-api");
81          final MavenProject spi = MavenTestUtils.getStub(
82                  "bundle", "se.jguru.foo.bar.spi.something", "bar-spi-something");
83          final MavenProject impl = MavenTestUtils.getStub(
84                  "bundle", "se.jguru.foo.bar.impl.something", "bar-impl-something");
85          final MavenProject test = MavenTestUtils.getStub("jar", "se.jguru.foo.bar.test.something", "bar-test");
86          final MavenProject poc = MavenTestUtils.getStub("bundle", "se.jguru.foo.bar.poc.something", "bar-poc");
87          final MavenProject war = MavenTestUtils.getStub("war", "se.jguru.foo.applications.bar", "bar-war");
88          final MavenProject ear = MavenTestUtils.getStub("ear", "se.jguru.foo.applications.bar", "bar-ear");
89          final MavenProject ejb = MavenTestUtils.getStub("ejb", "se.jguru.foo.applications.bar", "bar-ejb");
90          final MavenProject codestyle = MavenTestUtils.getStub("jar", "se.jguru.foo.codestyle", "bar-codestyle");
91          final MavenProject javaAgent = MavenTestUtils.getStub("bundle", "se.jguru.foo.bar.agent", "bar-agent");
92          final MavenProject standaloneApplication1
93                  = MavenTestUtils.getStub("bundle", "se.jguru.foo.bar.application", "bar-application");
94          final MavenProject standaloneApplication2
95                  = MavenTestUtils.getStub("jar", "se.jguru.foo.bar.application", "bar-application");
96          final MavenProject itest = MavenTestUtils.getStub("jar", "se.jguru.foo.it.bar", "bar-it");
97  
98          // Act & Assert
99          Assert.assertEquals(ProjectType.PARENT, ProjectType.getProjectType(parent));
100         Assert.assertEquals(ProjectType.REACTOR, ProjectType.getProjectType(reactor));
101         Assert.assertEquals(ProjectType.ASSEMBLY, ProjectType.getProjectType(assembly));
102         Assert.assertEquals(ProjectType.MODEL, ProjectType.getProjectType(model));
103         Assert.assertEquals(ProjectType.API, ProjectType.getProjectType(api));
104         Assert.assertEquals(ProjectType.SPI, ProjectType.getProjectType(spi));
105         Assert.assertEquals(ProjectType.IMPLEMENTATION, ProjectType.getProjectType(impl));
106         Assert.assertEquals(ProjectType.TEST, ProjectType.getProjectType(test));
107         Assert.assertEquals(ProjectType.PROOF_OF_CONCEPT, ProjectType.getProjectType(poc));
108         Assert.assertEquals(ProjectType.JEE_APPLICATION, ProjectType.getProjectType(war));
109         Assert.assertEquals(ProjectType.JEE_APPLICATION, ProjectType.getProjectType(ear));
110         Assert.assertEquals(ProjectType.JEE_APPLICATION, ProjectType.getProjectType(ejb));
111         Assert.assertEquals(ProjectType.JAVA_AGENT, ProjectType.getProjectType(javaAgent));
112         Assert.assertEquals(ProjectType.STANDALONE_APPLICATION, ProjectType.getProjectType(standaloneApplication1));
113         Assert.assertEquals(ProjectType.STANDALONE_APPLICATION, ProjectType.getProjectType(standaloneApplication2));
114         Assert.assertEquals(ProjectType.CODESTYLE, ProjectType.getProjectType(codestyle));
115         Assert.assertEquals(ProjectType.INTEGRATION_TEST, ProjectType.getProjectType(itest));
116     }

118     @Test
119     public void validateModelProjectPatterns() {
120 
121         // Act & Assert
122         Assert.assertTrue(ProjectType.MODEL.isCompliantArtifactID("test-foo-model"));
123         Assert.assertFalse(ProjectType.MODEL.isCompliantArtifactID("model-test"));
124         Assert.assertFalse(ProjectType.MODEL.isCompliantArtifactID("foo-model-test"));
125 
126         Assert.assertTrue(ProjectType.MODEL.isCompliantGroupID("test.foo.model"));
127         Assert.assertFalse(ProjectType.MODEL.isCompliantGroupID("model.test"));
128         Assert.assertFalse(ProjectType.MODEL.isCompliantGroupID("foo.model.test"));
129 
130         Assert.assertFalse(ProjectType.MODEL.isCompliantPackaging("pom"));
131         Assert.assertTrue(ProjectType.MODEL.isCompliantPackaging("jar"));
132         Assert.assertTrue(ProjectType.MODEL.isCompliantPackaging("bundle"));
133         Assert.assertFalse(ProjectType.MODEL.isCompliantPackaging("war"));
134     }
