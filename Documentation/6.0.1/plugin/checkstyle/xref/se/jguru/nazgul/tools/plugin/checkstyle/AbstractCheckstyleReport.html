459      * {@inheritDoc}
460      */
461     public void executeReport(Locale locale)
462             throws MavenReportException {
463         locator.addSearchPath(FileResourceLoader.ID, project.getFile().getParentFile().getAbsolutePath());
464         locator.addSearchPath("url", "");
465 
466         locator.setOutputDirectory(new File(project.getBuild().getDirectory()));
467 
468         // for when we start using maven-shared-io and maven-shared-monitor...
469         // locator = new Locator( new MojoLogMonitorAdaptor( getLog() ) );
470 
471         // locator = new Locator( getLog(), new File( project.getBuild().getDirectory() ) );
472 
473         ClassLoader currentClassLoader = Thread.currentThread().getContextClassLoader();
474 
475         try {
476             CheckstyleExecutorRequest request = createRequest()
477                     .setLicenseArtifacts(collectArtifacts("license"))
478                     .setConfigurationArtifacts(collectArtifacts("configuration"));
479 
480             CheckstyleResults results = checkstyleExecutor.executeCheckstyle(request);
481 
482             ResourceBundle bundle = getBundle(locale);
483             generateReportStatics();
484             generateMainReport(results, bundle);
485             if (enableRSS) {
486                 CheckstyleRssGeneratorRequest checkstyleRssGeneratorRequest =
487                         new CheckstyleRssGeneratorRequest(this.project, this.getCopyright(), outputDirectory, getLog());
488                 checkstyleRssGenerator.generateRSS(results, checkstyleRssGeneratorRequest);
489             }
490 
491         } catch (CheckstyleException e) {
492             throw new MavenReportException("Failed during checkstyle configuration", e);
493         } catch (CheckstyleExecutorException e) {
494             throw new MavenReportException("Failed during checkstyle execution", e);
495         } finally {
496             //be sure to restore original context classloader
497             Thread.currentThread().setContextClassLoader(currentClassLoader);
498         }
499     }

510     @SuppressWarnings("unchecked")
511     private List<Artifact> collectArtifacts(String hint) {
512         if (plugin == null || plugin.getGroupId() == null) {
513             // Maven 2.x workaround
514             plugin = mojoExecution.getMojoDescriptor().getPluginDescriptor();
515         }
516 
517         List<Artifact> artifacts = new ArrayList<>();
518 
519         PluginManagement pluginManagement = project.getBuild().getPluginManagement();
520         if (pluginManagement != null) {
521             artifacts.addAll(getCheckstylePluginDependenciesAsArtifacts(pluginManagement.getPluginsAsMap(), hint));
522         }
523 
524         artifacts.addAll(getCheckstylePluginDependenciesAsArtifacts(project.getBuild().getPluginsAsMap(), hint));
525 
526         return artifacts;
527     }
528 
529     private List<Artifact> getCheckstylePluginDependenciesAsArtifacts(Map<String, Plugin> plugins, String hint) {
530         List<Artifact> artifacts = new ArrayList<>();
531 
532         Plugin checkstylePlugin = plugins.get(plugin.getGroupId() + ":" + plugin.getArtifactId());
533         if (checkstylePlugin != null) {
534             for (Dependency dep : checkstylePlugin.getDependencies()) {
535                 // @todo if we can filter on hints, it should be done here...
536                 String depKey = dep.getGroupId() + ":" + dep.getArtifactId();
537                 final Artifact artifact = (Artifact) plugin.getArtifactMap().get(depKey);
538 
539                 if (artifact != null) {
540                     artifacts.add(artifact);
541                 } else if (getLog().isDebugEnabled()) {
542                     getLog().debug("Got null CheckstylePlugin Dependency Artifact from key ["
543                             + depKey + "]. Excluding null Artifact from return List [" + hint + "]. ");
544                 }
545             }
546         }
547         return artifacts;
548     }

550     /**
551      * Creates and returns the report generation listener.
552      *
553      * @return The audit listener.
554      * @throws MavenReportException If something goes wrong.
555      */
556     protected AuditListener getListener()
557             throws MavenReportException {
558         AuditListener listener = null;
559 
560         if (StringUtils.isNotEmpty(outputFileFormat)) {
561             File resultFile = outputFile;
562 
563             OutputStream out = getOutputStream(resultFile);
564 
565             if ("xml".equals(outputFileFormat)) {
566                 listener = new XMLLogger(out, true);
567             } else if ("plain".equals(outputFileFormat)) {
568                 listener = new DefaultLogger(out, true);
569             } else {
570                 // TODO: failure if not a report
571                 throw new MavenReportException("Invalid output file format: (" + outputFileFormat
572                         + "). Must be 'plain' or 'xml'.");
573             }
574         }
575 
576         return listener;
577     }

579     private OutputStream getOutputStream(File file)
580             throws MavenReportException {
581         File parentFile = file.getAbsoluteFile().getParentFile();
582 
583         if (!parentFile.exists()) {
584             parentFile.mkdirs();
585         }
586 
587         FileOutputStream fileOutputStream;
588         try {
589             fileOutputStream = new FileOutputStream(file);
590         } catch (FileNotFoundException e) {
591             throw new MavenReportException("Unable to create output stream: " + file, e);
592         }
593         return fileOutputStream;
594     }

629     private String getCopyright() {
630         String copyright;
631         int currentYear = Calendar.getInstance().get(Calendar.YEAR);
632         if (StringUtils.isNotEmpty(project.getInceptionYear())
633                 && !String.valueOf(currentYear).equals(project.getInceptionYear())) {
634             copyright = project.getInceptionYear() + " - " + currentYear;
635         } else {
636             copyright = String.valueOf(currentYear);
637         }
638 
639         if ((project.getOrganization() != null) && StringUtils.isNotEmpty(project.getOrganization().getName())) {
640             copyright = copyright + " " + project.getOrganization().getName();
641         }
642         return copyright;
643     }

645     private void generateMainReport(CheckstyleResults results, ResourceBundle bundle) {
646         CheckstyleReportGenerator generator =
647                 new CheckstyleReportGenerator(getSink(), bundle, project.getBasedir(), siteTool, configLocation);
648 
649         generator.setLog(getLog());
650         generator.setEnableRulesSummary(enableRulesSummary);
651         generator.setEnableSeveritySummary(enableSeveritySummary);
652         generator.setEnableFilesSummary(enableFilesSummary);
653         generator.setEnableRSS(enableRSS);
654         generator.setCheckstyleConfig(results.getConfiguration());
655         if (linkXRef) {
656             String relativePath = PathTool.getRelativePath(getOutputDirectory(), xrefLocation.getAbsolutePath());
657             if (StringUtils.isEmpty(relativePath)) {
658                 relativePath = ".";
659             }
660             relativePath = relativePath + "/" + xrefLocation.getName();
661             if (xrefLocation.exists()) {
662                 // XRef was already generated by manual execution of a lifecycle
663                 // binding
664                 generator.setXrefLocation(relativePath);
665             } else {
666                 // Not yet generated - check if the report is on its way
667                 for (ReportPlugin report : (Iterable<ReportPlugin>) getProject().getReportPlugins()) {
668                     String artifactId = report.getArtifactId();
669                     if ("maven-jxr-plugin".equals(artifactId) || "jxr-maven-plugin".equals(artifactId)) {
670                         generator.setXrefLocation(relativePath);
671                     }
672                 }
673             }
674 
675             if (generator.getXrefLocation() == null && results.getFileCount() > 0) {
676                 getLog().warn("Unable to locate Source XRef to link to - DISABLED");
677             }
678         }
679         if (treeWalkerNames != null) {
680             generator.setTreeWalkerNames(treeWalkerNames);
681         }
682         generator.generateReport(results);
683     }

689     protected List<File> getSourceDirectories() {
690         List<File> sourceDirs = null;
691         // if sourceDirectory is explicitly set, use it
692         if (sourceDirectory != null) {
693             sourceDirs = Collections.singletonList(sourceDirectory);
694         } else {
695             sourceDirs = new ArrayList<>(sourceDirectories.size());
696             for (String sourceDir : sourceDirectories) {
697                 sourceDirs.add(FileUtils.resolveFile(project.getBasedir(), sourceDir));
698             }
699         }
700 
701         return sourceDirs;
702     }

704     protected List<File> getTestSourceDirectories() {
705         List<File> testSourceDirs = null;
706         // if testSourceDirectory is explicitly set, use it
707         if (testSourceDirectory != null) {
708             testSourceDirs = Collections.singletonList(testSourceDirectory);
709         }
710         // probably null-check only required due to MavenProjectStubs
711         else if (testSourceDirectories != null) {
712             testSourceDirs = new ArrayList<>(testSourceDirectories.size());
713             for (String testSourceDir : testSourceDirectories) {
714                 testSourceDirs.add(FileUtils.resolveFile(project.getBasedir(), testSourceDir));
715             }
716         }

718         return testSourceDirs;
