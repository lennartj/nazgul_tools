486     /**
487      * {@inheritDoc}
488      */
489     public void execute()
490             throws MojoExecutionException, MojoFailureException {
491         if (skip) {
492             return;
493         }
494 
495         if (!skipExec) {
496             if (checkstyleRules != null) {
497                 if (!"sun_checks.xml".equals(configLocation)) {
498                     throw new MojoExecutionException("If you use inline configuration for rules, don't specify "
499                             + "a configLocation");
500                 }
501                 if (checkstyleRules.getChildCount() > 1) {
502                     throw new MojoExecutionException("Currently only one root module is supported");
503                 }
504 
505                 PlexusConfiguration checkerModule = checkstyleRules.getChild(0);
506 
507                 try {
508                     FileUtils.forceMkdir(rulesFiles.getParentFile());
509                     FileUtils.fileWrite(rulesFiles, CHECKSTYLE_FILE_HEADER + checkerModule.toString());
510                 } catch (final IOException e) {
511                     throw new MojoExecutionException(e.getMessage(), e);
512                 }
513                 configLocation = rulesFiles.getAbsolutePath();
514             }
515 
516             ClassLoader currentClassLoader = Thread.currentThread().getContextClassLoader();
517 
518             try {
519                 CheckstyleExecutorRequest request = new CheckstyleExecutorRequest();
520                 request.setConsoleListener(getConsoleListener()).setConsoleOutput(consoleOutput)
521                         .setExcludes(excludes).setFailsOnError(failsOnError).setIncludes(includes)
522                         .setResourceIncludes(resourceIncludes)
523                         .setResourceExcludes(resourceExcludes)
524                         .setIncludeResources(includeResources)
525                         .setIncludeTestResources(includeTestResources)
526                         .setIncludeTestSourceDirectory(includeTestSourceDirectory).setListener(getListener())
527                         .setProject(project).setSourceDirectories(getSourceDirectories())
528                         .setResources(resources).setTestResources(testResources)
529                         .setStringOutputStream(stringOutputStream).setSuppressionsLocation(suppressionsLocation)
530                         .setTestSourceDirectories(getTestSourceDirectories()).setConfigLocation(configLocation)
531                         .setConfigurationArtifacts(collectArtifacts("config"))
532                         .setPropertyExpansion(propertyExpansion)
533                         .setHeaderLocation(headerLocation).setLicenseArtifacts(collectArtifacts("license"))
534                         .setCacheFile(cacheFile).setSuppressionsFileExpression(suppressionsFileExpression)
535                         .setEncoding(encoding).setPropertiesLocation(propertiesLocation);
536                 checkstyleExecutor.executeCheckstyle(request);
537 
538             } catch (CheckstyleException e) {
539                 throw new MojoExecutionException("Failed during checkstyle configuration", e);
540             } catch (CheckstyleExecutorException e) {
541                 throw new MojoExecutionException("Failed during checkstyle execution", e);
542             } finally {
543                 //be sure to restore original context classloader
544                 Thread.currentThread().setContextClassLoader(currentClassLoader);
545             }
546         }
547 
548         if (!"xml".equals(outputFileFormat)) {
549             throw new MojoExecutionException("Output format is '" + outputFileFormat
550                     + "', checkstyle:check requires format to be 'xml'.");
551         }
552 
553         if (!outputFile.exists()) {
554             getLog().info("Unable to perform checkstyle:check, unable to find checkstyle:checkstyle outputFile.");
555             return;
556         }
557 
558         Reader reader = null;
559         try {
560             reader = new BufferedReader(ReaderFactory.newXmlReader(outputFile));
561 
562             XmlPullParser xpp = new MXParser();
563             xpp.setInput(reader);
564 
565             int violations = countViolations(xpp);
566 
567             if (violations > maxAllowedViolations) {
568                 if (failOnViolation) {
569                     String msg =
570                             "You have " + violations + " Checkstyle violation" + ((violations > 1) ? "s" : "") + ".";
571                     if (maxAllowedViolations > 0) {
572                         msg += " The maximum number of allowed violations is " + maxAllowedViolations + ".";
573                     }
574                     throw new MojoFailureException(msg);
575                 }
576 
577                 getLog().warn("checkstyle:check violations detected but failOnViolation set to false");
578             }
579 
580             reader.close();
581             reader = null;
582         } catch (IOException | XmlPullParserException e) {
583             throw new MojoExecutionException("Unable to read Checkstyle results xml: " + outputFile.getAbsolutePath(),
584                     e);
585         } finally {
586             IOUtil.close(reader);
587         }
588     }

590     private int countViolations(XmlPullParser xpp)
591             throws XmlPullParserException, IOException {
592         int count = 0;
593         int ignoreCount = 0;
594         RuleUtil.Matcher[] ignores =
595                 (violationIgnore == null) ? null : RuleUtil.parseMatchers(violationIgnore.split(","));
596 
597         String basedir = project.getBasedir().getAbsolutePath();
598         String file = "";
599         for (int eventType = xpp.getEventType(); eventType != XmlPullParser.END_DOCUMENT; eventType = xpp.next()) {
600             if (eventType != XmlPullParser.START_TAG) {
601                 continue;
602             } else if ("file".equals(xpp.getName())) {
603                 file = PathTool.getRelativeFilePath(basedir, xpp.getAttributeValue("", "name"));
604                 //file = file.substring( file.lastIndexOf( File.separatorChar ) + 1 );
605             } else if ("error".equals(xpp.getName())) {
606                 String severity = xpp.getAttributeValue("", "severity");
607 
608                 if (!isViolation(severity)) {
609                     continue;
610                 }
611 
612                 String source = xpp.getAttributeValue("", "source");
613 
614                 if (ignore(ignores, source)) {
615                     ignoreCount++;
616                 } else {
617                     count++;
618 
619                     if (logViolationsToConsole) {
620                         String line = xpp.getAttributeValue("", "line");
621                         String column = xpp.getAttributeValue("", "column");
622                         String message = xpp.getAttributeValue("", "message");
623                         String rule = RuleUtil.getName(source);
624                         String category = RuleUtil.getCategory(source);
625 
626                         log(severity, file + ":[" + line + ((column == null) ? "" : (',' + column)) + "] ("
627                                 + category + ") " + rule + ": " + message);
628                     }
629                 }
630             }
631         }
632 
633         if (ignoreCount > 0) {
634             getLog().info("Ignored " + ignoreCount + " error" + ((ignoreCount > 1) ? "s" : "") + ", " + count
635                     + " violation" + ((count > 1) ? "s" : "") + " remaining.");
636         }
637 
638         return count;
639     }

641     private void log(String severity, String message) {
642         if ("info".equals(severity)) {
643             getLog().info(message);
644         } else if ("warning".equals(severity)) {
645             getLog().warn(message);
646         } else {
647             getLog().error(message);
648         }
649     }

651     /**
652      * Checks if the given severity is considered a violation.
653      *
654      * @param severity The severity to check
655      * @return <code>true</code> if the given severity is a violation, otherwise <code>false</code>
656      */
657     private boolean isViolation(String severity) {
658         if ("error".equals(severity)) {
659             return "error".equals(violationSeverity) || "warning".equals(violationSeverity)
660                     || "info".equals(violationSeverity);
661         } else if ("warning".equals(severity)) {
662             return "warning".equals(violationSeverity) || "info".equals(violationSeverity);
663         } else if ("info".equals(severity)) {
664             return "info".equals(violationSeverity);
665         } else {
666             return false;
667         }
668     }
669 
670     private boolean ignore(RuleUtil.Matcher[] ignores, String source) {
671         if (ignores != null) {
672             for (RuleUtil.Matcher ignore : ignores) {
673                 if (ignore.match(source)) {
674                     return true;
675                 }
676             }
677         }
678 
679         return false;
680     }

682     private DefaultLogger getConsoleListener()
683             throws MojoExecutionException {
684         DefaultLogger consoleListener;
685 
686         if (useFile == null) {
687             stringOutputStream = new ByteArrayOutputStream();
688             consoleListener = new DefaultLogger(stringOutputStream, false);
689         } else {
690             OutputStream out = getOutputStream(useFile);
691 
692             consoleListener = new DefaultLogger(out, true);
693         }
694 
695         return consoleListener;
696     }

698     private OutputStream getOutputStream(File file)
699             throws MojoExecutionException {
700         File parentFile = file.getAbsoluteFile().getParentFile();
701 
702         if (!parentFile.exists()) {
703             parentFile.mkdirs();
704         }
705 
706         FileOutputStream fileOutputStream;
707         try {
708             fileOutputStream = new FileOutputStream(file);
709         } catch (FileNotFoundException e) {
710             throw new MojoExecutionException("Unable to create output stream: " + file, e);
711         }
712         return fileOutputStream;
713     }

715     private AuditListener getListener()
716             throws MojoFailureException, MojoExecutionException {
717         AuditListener listener = null;
718 
719         if (StringUtils.isNotEmpty(outputFileFormat)) {
720             File resultFile = outputFile;
721 
722             OutputStream out = getOutputStream(resultFile);
723 
724             if ("xml".equals(outputFileFormat)) {
725                 listener = new XMLLogger(out, true);
726             } else if ("plain".equals(outputFileFormat)) {
727                 listener = new DefaultLogger(out, true);
728             } else {
729                 throw new MojoFailureException("Invalid output file format: (" + outputFileFormat
730                         + "). Must be 'plain' or 'xml'.");
731             }
732         }
733 
734         return listener;
735     }

737     @SuppressWarnings("unchecked")
738     private List<Artifact> collectArtifacts(String hint) {
739         if (plugin == null || plugin.getGroupId() == null) {
740             // Maven 2.x workaround
741             plugin = mojoExecution.getMojoDescriptor().getPluginDescriptor();
742         }
743 
744         List<Artifact> artifacts = new ArrayList<>();
745 
746         PluginManagement pluginManagement = project.getBuild().getPluginManagement();
747         if (pluginManagement != null) {
748             artifacts.addAll(getCheckstylePluginDependenciesAsArtifacts(pluginManagement.getPluginsAsMap(), hint));
749         }
750 
751         artifacts.addAll(getCheckstylePluginDependenciesAsArtifacts(project.getBuild().getPluginsAsMap(), hint));
752 
753         return artifacts;
754     }

756     private List<Artifact> getCheckstylePluginDependenciesAsArtifacts(Map<String, Plugin> plugins, String hint) {
757         List<Artifact> artifacts = new ArrayList<>();
758 
759         Plugin checkstylePlugin = plugins.get(plugin.getGroupId() + ":" + plugin.getArtifactId());
760         if (checkstylePlugin != null) {
761             for (Dependency dep : checkstylePlugin.getDependencies()) {
762                 // @todo if we can filter on hints, it should be done here...
763                 String depKey = dep.getGroupId() + ":" + dep.getArtifactId();
764                 artifacts.add((Artifact) plugin.getArtifactMap().get(depKey));
765             }
766         }
767         return artifacts;
768     }
